 <script>

   // three.js: fragmented icosahedron frozen mid-explosion

   const container = document.getElementById('star-canvas');

   const W = 510, H = 250;

   const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

   renderer.setSize(W, H);

   renderer.setClearColor(0x060000, 1);

   container.appendChild(renderer.domElement);

   const scene = new THREE.Scene();

   const camera = new THREE.PerspectiveCamera(58, W / H, 0.1, 100);

   camera.position.set(0, 0, 4);

   scene.add(new THREE.AmbientLight(0x110000, 3));

   const redLight = new THREE.PointLight(0xff2200, 6, 10);

   redLight.position.set(1, 1, 2);

   scene.add(redLight);

   const orangeLight = new THREE.PointLight(0xff8800, 4, 8);

   orangeLight.position.set(-2, -1, 1);

   scene.add(orangeLight);

   // generate fragment shards from icosahedron faces

   const baseGeo = new THREE.IcosahedronGeometry(1.0, 1);

   const shards = [];

   const shardGroup = new THREE.Group();

   scene.add(shardGroup);

   // create individual triangle shards

   const pos = baseGeo.attributes.position;

   const count = pos.count;

   for (let i = 0; i < count; i += 3) {

     const v0 = new THREE.Vector3(pos.getX(i), pos.getY(i), pos.getZ(i));

     const v1 = new THREE.Vector3(pos.getX(i+1), pos.getY(i+1), pos.getZ(i+1));

     const v2 = new THREE.Vector3(pos.getX(i+2), pos.getY(i+2), pos.getZ(i+2));

     const center = new THREE.Vector3().addVectors(v0, v1).add(v2).divideScalar(3);

     const geo = new THREE.BufferGeometry();

     // offset verts relative to center

     const verts = new Float32Array([

       v0.x - center.x, v0.y - center.y, v0.z - center.z,

       v1.x - center.x, v1.y - center.y, v1.z - center.z,

       v2.x - center.x, v2.y - center.y, v2.z - center.z,

     ]);

     geo.setAttribute('position', new THREE.BufferAttribute(verts, 3));

     geo.computeVertexNormals();

     const shade = 0.5 + Math.random() * 0.5;

     const col = new THREE.Color(shade, shade * 0.2, 0);

     const mat = new THREE.MeshPhongMaterial({

       color: col,

       emissive: new THREE.Color(shade * 0.3, 0, 0),

       transparent: true,

       opacity: 0.8 + Math.random() * 0.2,

       shininess: 80,

       side: THREE.DoubleSide,

     });

     const mesh = new THREE.Mesh(geo, mat);

     mesh.position.copy(center);

     // each shard has a drift vector (frozen mid-explosion)

     const drift = center.clone().normalize().multiplyScalar(0.3 + Math.random() * 0.9);

     const spin = new THREE.Vector3(

       (Math.random() - 0.5) * 0.04,

       (Math.random() - 0.5) * 0.04,

       (Math.random() - 0.5) * 0.04

     );

     shards.push({ mesh, origin: center.clone(), drift, spin, phase: Math.random() * Math.PI * 2 });

     shardGroup.add(mesh);

   }

   // glowing core point

   const coreMat = new THREE.PointsMaterial({ color: 0xff4400, size: 0.12, transparent: true, opacity: 0.9 });

   const coreGeo = new THREE.BufferGeometry();

   coreGeo.setAttribute('position', new THREE.BufferAttribute(new Float32Array([0,0,0]), 3));

   const core = new THREE.Points(coreGeo, coreMat);

   scene.add(core);

   // particle debris

   const debrisCount = 200;

   const debrisPositions = new Float32Array(debrisCount * 3);

   const debrisVelocities = [];

   for (let i = 0; i < debrisCount; i++) {

     const theta = Math.random() * Math.PI * 2;

     const phi = Math.acos(2 * Math.random() - 1);

     const r = 0.5 + Math.random() * 2;

     debrisPositions[i*3] = r * Math.sin(phi) * Math.cos(theta);

     debrisPositions[i*3+1] = r * Math.sin(phi) * Math.sin(theta);

     debrisPositions[i*3+2] = r * Math.cos(phi);

     debrisVelocities.push(new THREE.Vector3(

       (Math.random() - 0.5) * 0.003,

       (Math.random() - 0.5) * 0.003,

       (Math.random() - 0.5) * 0.003

     ));

   }

   const debrisGeo = new THREE.BufferGeometry();

   debrisGeo.setAttribute('position', new THREE.BufferAttribute(debrisPositions, 3));

   const debrisMat = new THREE.PointsMaterial({ color: 0xff6600, size: 0.025, transparent: true, opacity: 0.5 });

   const debris = new THREE.Points(debrisGeo, debrisMat);

   scene.add(debris);

   let starMode = 'collapse';

   window.setStarMode = function(mode, btn) {

     starMode = mode;

     document.querySelectorAll('.star-btn').forEach(b => b.classList.remove('active'));

     btn.classList.add('active');

   };

   let elapsed = 0;

   function animate() {

     requestAnimationFrame(animate);

     const t = Date.now() * 0.001;

     elapsed++;

     let explodeAmt = 0;

     if (starMode === 'collapse') explodeAmt = 0.4 + 0.2 * Math.sin(t * 0.5);

     else if (starMode === 'nova') explodeAmt = 0.85 + 0.1 * Math.sin(t * 2);

     else explodeAmt = 0.15 + 0.05 * Math.sin(t * 0.3);

     // animate shards

     shards.forEach(s => {

       const pulse = Math.sin(t * 1.2 + s.phase) * 0.04;

       s.mesh.position.lerpVectors(s.origin, s.origin.clone().add(s.drift), explodeAmt + pulse);

       s.mesh.rotation.x += s.spin.x;

       s.mesh.rotation.y += s.spin.y;

       s.mesh.rotation.z += s.spin.z;

       s.mesh.material.opacity = starMode === 'remnant' ? 0.3 + Math.random() * 0.1 : 0.7 + Math.random() * 0.3;

     });

     shardGroup.rotation.y += 0.004;

     shardGroup.rotation.x += 0.002;

     // debris

     const dp = debrisGeo.attributes.position.array;

     for (let i = 0; i < debrisCount; i++) {

       dp[i*3]   += debrisVelocities[i].x;

       dp[i*3+1] += debrisVelocities[i].y;

       dp[i*3+2] += debrisVelocities[i].z;

       const dist = Math.sqrt(dp[i*3]**2 + dp[i*3+1]**2 + dp[i*3+2]**2);

       if (dist > 3.5) {

         dp[i*3] = (Math.random()-0.5)*0.3;

         dp[i*3+1] = (Math.random()-0.5)*0.3;

         dp[i*3+2] = (Math.random()-0.5)*0.3;

       }

     }

     debrisGeo.attributes.position.needsUpdate = true;

     // lights

     redLight.position.x = 2 * Math.sin(t * 0.7);

     redLight.position.y = 1.5 * Math.cos(t * 0.5);

     orangeLight.position.x = -2 * Math.cos(t * 0.9);

     orangeLight.intensity = starMode === 'nova' ? 6 + Math.sin(t * 3) * 2 : 4;

     // core glow

     coreMat.size = starMode === 'nova' ? 0.25 + Math.sin(t*4) * 0.1 : 0.08;

     renderer.render(scene, camera);

   }

   animate();

   // event log

   const collapseEvents = [

     ['ec', 'iron core collapse initiated'],

     ['em', 'neutrino burst detected: 3.2e53 erg'],

     ['ec', 'shock wave stalled at 200km radius'],

     ['em', 'convective overturn establishing'],

     ['ec', 'neutron star formation confirmed'],

     ['em', 'envelope ejection velocity: 10,000 km/s'],

     ['ec', 'r-process nucleosynthesis active'],

     ['em', 'Fe-56 peak at core — no further fusion'],

     ['ec', 'gravity wins at t+0.25s'],

     ['em', 'density: 2.3 x nuclear saturation'],

     ['ec', 'bounce shock generated at r=50km'],

     ['em', 'explosion energy: 1 foe (10^51 erg)'],

   ];

   let evtCount = 0;

   let logCount = 0;

   function addEvent() {

     const log = document.getElementById('obit-log');

     const entry = collapseEvents[evtCount % collapseEvents.length];

     evtCount++;

     logCount++;

     document.getElementById('event-count').textContent = logCount + ' events';

     const now = new Date();

     const ts = `T+${String(logCount).padStart(3,'0')}s`;

     const div = document.createElement('div');

     div.className = 'obit-entry';

     div.innerHTML = `<span class="et">[${ts}]</span> <span class="${entry[0]}">${entry[1]}</span>`;

     log.appendChild(div);

     log.scrollTop = log.scrollHeight;

     if (log.children.length > 40) log.removeChild(log.firstChild);

   }

   function pad(n) { return String(n).padStart(2,'0'); }

   let upSec = 0;

   setInterval(() => {

     upSec++;

     const h = Math.floor(upSec / 3600);

     const m = Math.floor((upSec % 3600) / 60);

     const s = upSec % 60;

     document.getElementById('star-clock').textContent = `${pad(h)}:${pad(m)}:${pad(s)}`;

     // jitter data

     document.getElementById('mass-val').textContent = (16 + Math.random()*4).toFixed(1) + 'M&#x2299;';

     document.getElementById('radius-val').textContent = (3 + Math.random()*2).toFixed(1) + 'R&#x2299;';

     document.getElementById('temp-val').textContent = Math.floor(40 + Math.random()*15) + 'kK';

     document.getElementById('collapse-pct').textContent = 'COL: ' + Math.floor(80 + Math.random()*19) + '%';

     // coordinates drift

     const raH = 5, raM = 34, raS = Math.floor(Math.random()*60);

     document.getElementById('ra-val').textContent = `${raH}h ${raM}m ${pad(raS)}s`;

   }, 1000);

   setInterval(addEvent, 2800);

   addEvent();
