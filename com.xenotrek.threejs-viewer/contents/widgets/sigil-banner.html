           revealEngravings() {

               this.hiddenEngravings.forEach((engraving, index) => {

                   setTimeout(() => {

                       this.animateProperty(engraving.material, 'emissiveIntensity', 0.4, 300);

                       this.animateProperty(engraving.material, 'opacity', 0.8, 300);

                   }, index * 50);

               });

           }

           hideEngravings() {

               this.hiddenEngravings.forEach((engraving, index) => {

                   setTimeout(() => {

                       this.animateProperty(engraving.material, 'emissiveIntensity', 0.0, 200);

                       this.animateProperty(engraving.material, 'opacity', 0.0, 200);

                   }, index * 30);

               });

           }

           animateProperty(object, property, targetValue, duration) {

               const startValue = object[property];

               const startTime = Date.now();

               const animate = () => {

                   const elapsed = Date.now() - startTime;

                   const progress = Math.min(elapsed / duration, 1);

                   const eased = this.easeInOutCubic(progress);

                   object[property] = startValue + (targetValue - startValue) * eased;

                   if (progress < 1) {

                       requestAnimationFrame(animate);

                   }

               };

               animate();

           }

           easeInOutCubic(t) {

               return t < 0.5 ? 4 * t * t * t : (t - 1) * (2 * t - 2) * (2 * t - 2) + 1;

           }

               spark.style.top = y + 'px';

               // Random direction for spark movement

               const dx = (Math.random() - 0.5) * 50 + 'px';

               const dy = (Math.random() - 0.5) * 50 + 'px';

               spark.style.setProperty('--dx', dx);

               spark.style.setProperty('--dy', dy);

               this.container.appendChild(spark);

               // Remove spark after animation

               setTimeout(() => {

                   if (spark.parentNode) {

                       spark.parentNode.removeChild(spark);

                   }

               }, 2000);

           }

           handleResize() {

               const width = this.container.offsetWidth;

               const height = this.container.offsetHeight;

               this.camera.aspect = width / height;

               this.camera.updateProjectionMatrix();

               this.renderer.setSize(width, height);

           }

           animate() {

               requestAnimationFrame(() => this.animate());

               this.time += 0.016;

               this.sparkTimer += 0.016;

               // Multi-layered rotation system with depth-based speeds

               const baseRotation = this.isHovered ? 0.025 : 0.008;

               this.sigilGroup.rotation.z += baseRotation;

               // Animate primary spike arrays

               this.sigilSpikes.forEach((spike, index) => {

                   const offset = index * 0.5;

                   const intensity = 0.4 + Math.sin(this.time * 2.5 + offset) * 0.3;

                   spike.material.emissiveIntensity = intensity;

                   spike.position.z = 0.45 + Math.sin(this.time * 1.5 + offset) * 0.05;

               });

               // Animate secondary spikes with different frequency

               this.secondarySpikes.forEach((spike, index) => {

                   const offset = index * 0.7;

                   const intensity = 0.3 + Math.sin(this.time * 3 + offset) * 0.2;

                   spike.material.emissiveIntensity = intensity;

                   spike.rotation.y = Math.sin(this.time + offset) * 0.2;

               });

               // Floating elements orbital motion

               this.floatingElements.forEach((element, index) => {

                   const orbitSpeed = 0.01 + (index * 0.002);

                   const orbitRadius = 1.6;

                   const baseAngle = (index / 6) * Math.PI * 2;

                   const currentAngle = baseAngle + this.time * orbitSpeed;

                   element.position.x = Math.cos(currentAngle) * orbitRadius;

                   element.position.y = Math.sin(currentAngle) * orbitRadius;

                   element.position.z = 0.6 + Math.sin(this.time * 2 + index) * 0.3;

                   element.rotation.x += 0.02;

                   element.rotation.y += 0.015;

                   const pulseIntensity = 0.2 + Math.sin(this.time * 4 + index) * 0.1;

                   element.material.emissiveIntensity = pulseIntensity;

               });

               // Dynamic lighting system

               this.redLight.intensity = 1.2 + Math.sin(this.time * 1.8) * 0.4;

               this.redLight.position.x = Math.sin(this.time * 0.5) * 0.5;

               this.accentLight1.intensity = 0.4 + Math.sin(this.time * 2.2) * 0.2;

               this.accentLight2.intensity = 0.3 + Math.cos(this.time * 1.7) * 0.15;

               this.rimLight.intensity = 0.8 + Math.sin(this.time * 3) * 0.3;

               this.depthLight1.position.x = 2 + Math.sin(this.time * 0.8) * 0.5;

               this.depthLight1.intensity = 0.5 + Math.cos(this.time * 2.5) * 0.2;

               this.depthLight2.position.y = Math.cos(this.time * 0.6) * 0.8;

               this.depthLight2.intensity = 0.4 + Math.sin(this.time * 2.8) * 0.1;

               // Enhanced spark generation with depth awareness

               if (this.sparkTimer > 0.8 + Math.random() * 1.5) {

                   this.createSpark();

                   this.sparkTimer = 0;

               }

               // Aggressive glitch effects on multiple elements

               if (Math.random() < 0.025) {

                   this.sigilPlate.material.emissiveIntensity = 0.8;

                   setTimeout(() => {

                       this.sigilPlate.material.emissiveIntensity = 0.15;

                   }, 80);

               }

               // Secondary glitch on floating elements

               if (Math.random() < 0.015) {

                   const randomElement = this.floatingElements[Math.floor(Math.random() * this.floatingElements.length)];

                   randomElement.material.emissiveIntensity = 0.9;

                   setTimeout(() => {

                       randomElement.material.emissiveIntensity = 0.2;

                   }, 120);

               }

               // Depth-based camera subtle movement for parallax effect

               this.camera.position.x = Math.sin(this.time * 0.3) * 0.1;

               this.camera.position.y = Math.cos(this.time * 0.4) * 0.08;

               this.renderer.render(this.scene, this.camera);

           }

       }

           new SigilBanner('canvasContainer');

       });
