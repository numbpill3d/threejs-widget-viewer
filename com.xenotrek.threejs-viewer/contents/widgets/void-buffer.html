       // NEURAL PARAMETERS

       const SYNAPSE_COLOR = 0x0F1525;

       const AXON_COLOR = 0xFF1A1A;

       const MEMBRANE_COLOR = 0x000000;

       const RECEPTOR_COLOR = 0x151A20;

       const PARTICLE_COUNT = 150;

       // SYSTEM INITIALIZATION

       let scene, camera, renderer;

       let nervousSystem = [];

       let particles = [];

       let organicLayers = [];

       let syntheticTime = 0;

       let container = document.getElementById('neural-interface');

       let synapticResponse = document.getElementById('synaptic-response');

       // NEUROMORPHIC INTERFACE INITIALIZATION

       function initializeNeuralInterface() {

           // Membrane formation

           scene = new THREE.Scene();

           scene.background = new THREE.Color(MEMBRANE_COLOR);

           // Visual cortex parameters

           camera = new THREE.PerspectiveCamera(60, container.clientWidth / container.clientHeight, 0.1, 1000);

           camera.position.set(0, 15, 40);

           camera.lookAt(0, 0, 0);

           // Retinal projection system

           renderer = new THREE.WebGLRenderer({ antialias: false, alpha: true });

           renderer.setSize(container.clientWidth, container.clientHeight);

           renderer.setPixelRatio(window.devicePixelRatio * 0.7);

           container.appendChild(renderer.domElement);

           // Basal illumination

           const ambientLight = new THREE.AmbientLight(0x050708);

           scene.add(ambientLight);

           // Directed stimulus

           const directionalLight = new THREE.DirectionalLight(0xccddff, 0.4);

           directionalLight.position.set(1, 1, 1);

           scene.add(directionalLight);

           // Point light for organic glow

           const pointLight = new THREE.PointLight(0xff3333, 0.8, 50);

           pointLight.position.set(0, 10, 20);

           scene.add(pointLight);

           // Generate neural substrate

           generateSubstrate();

           // Create organic layering

           createOrganicLayers();

           // Spawn particle system

           createParticleSystem();

           // Generate symbiotic buttons

           createSymbioticButtons();

           // Bind synaptic receptors

           container.addEventListener('mousemove', stimulateSynapse);

           container.addEventListener('click', triggerActionPotential);

           window.addEventListener('resize', recalibrateVisualField);

       }

       // SUBSTRATE GENERATION

       function generateSubstrate() {

           // Add central spinal structure

           const spineGeometry = new THREE.BoxGeometry(1, 1, 60);

           const spineMaterial = new THREE.MeshBasicMaterial({

               color: 0x333333,

               wireframe: true

           });

           const spine = new THREE.Mesh(spineGeometry, spineMaterial);

           spine.position.y = -5;

           scene.add(spine);

           // Generate digital neural networks

           for (let i = 0; i < 48; i++) {

               const nodeSize = Math.random() * 0.8 + 0.2;

               // Create axon geometry

               let geometry;

               const morphType = Math.floor(Math.random() * 4);

               if (morphType === 0) {

                   geometry = new THREE.TetrahedronGeometry(nodeSize * 2);

               } else if (morphType === 1) {

                   geometry = new THREE.OctahedronGeometry(nodeSize * 1.5);

               } else if (morphType === 2) {

                   geometry = new THREE.IcosahedronGeometry(nodeSize);

               } else {

                   geometry = new THREE.BoxGeometry(nodeSize, nodeSize, nodeSize);

               }

               // Myelin sheath material

               const material = new THREE.MeshBasicMaterial({

                   color: SYNAPSE_COLOR,

                   wireframe: true,

                   transparent: true,

                   opacity: 0.8

               });

               // Generate neuron

               const neuron = new THREE.Mesh(geometry, material);

               // Position within neural field

               neuron.position.x = Math.random() * 40 - 20;

               neuron.position.y = Math.random() * 10 - 5;

               neuron.position.z = Math.random() * 40 - 20;

               // Random rotation to simulate chaotic neural growth

               neuron.rotation.x = Math.random() * Math.PI;

               neuron.rotation.y = Math.random() * Math.PI;

               neuron.rotation.z = Math.random() * Math.PI;

               // Add to neural network

               scene.add(neuron);

               nervousSystem.push(neuron);

               // Create dendrite connections

               if (i > 0 && Math.random() > 0.3) {

                   const previousNeuron = nervousSystem[Math.floor(Math.random() * i)];

                   connectNeurons(neuron, previousNeuron);

               }

           }

           // Create neural substrate

           const substrateGeometry = new THREE.PlaneGeometry(80, 80, 20, 20);

           substrateGeometry.rotateX(-Math.PI / 2);

           // Deform to create sulci and gyri

           const vertices = substrateGeometry.attributes.position.array;

           for (let i = 0; i < vertices.length; i += 3) {

               vertices[i + 2] = Math.sin(vertices[i] / 5) * Math.cos(vertices[i + 1] / 5) * 2;

           }

           const substrateMaterial = new THREE.MeshBasicMaterial({

               color: RECEPTOR_COLOR,

               wireframe: true,

               transparent: true,

               opacity: 0.3

           });

           const substrate = new THREE.Mesh(substrateGeometry, substrateMaterial);

           substrate.position.y = -8;

           scene.add(substrate);

       }

       // CREATE ORGANIC MEMBRANE LAYERING

       function createOrganicLayers() {

           // Create multiple layered neural membranes

           for (let i = 0; i < 3; i++) {

               const layerGeometry = new THREE.PlaneGeometry(100 - i * 10, 30, 20, 5);

               // Create tissue folds

               const vertices = layerGeometry.attributes.position.array;

               for (let j = 0; j < vertices.length; j += 3) {

                   vertices[j + 2] = Math.sin(vertices[j] / 3) * 0.5 + Math.cos(vertices[j + 1] * 2) * 0.5;

               }

               // Create semi-transparent tissue material

               const layerMaterial = new THREE.MeshBasicMaterial({

                   color: new THREE.Color(0.1 + i * 0.05, 0.05, 0.05),

                   wireframe: true,

                   transparent: true,

                   opacity: 0.3 - i * 0.05

               });

               const layer = new THREE.Mesh(layerGeometry, layerMaterial);

               layer.position.z = -10 + i * 5;

               layer.position.y = 5 - i * 2;

               layer.rotation.x = Math.PI / 2.5;

               scene.add(layer);

               organicLayers.push(layer);

           }

           // Create central sphere as neural core

           const coreGeometry = new THREE.SphereGeometry(5, 8, 8);

           const coreMaterial = new THREE.MeshBasicMaterial({

               color: 0x330000,

               wireframe: true,

               transparent: true,

               opacity: 0.4

           });

           const core = new THREE.Mesh(coreGeometry, coreMaterial);

           core.position.z = 0;

           core.position.y = 0;

           scene.add(core);

           organicLayers.push(core);

           // Create surrounding torus as protective sheathing

           const torusGeometry = new THREE.TorusGeometry(12, 1, 8, 16);

           const torusMaterial = new THREE.MeshBasicMaterial({

               color: 0x220000,

               wireframe: true,

               transparent: true,

               opacity: 0.5

           });

           const torus = new THREE.Mesh(torusGeometry, torusMaterial);

           torus.rotation.x = Math.PI / 2;

           scene.add(torus);

           organicLayers.push(torus);

       }

       // CREATE PARTICLE SYSTEM FOR NEURAL TRANSMISSION

       function createParticleSystem() {

           // Create particle system for neural transmitters

           const particleGeometry = new THREE.BufferGeometry();

           const particlePositions = new Float32Array(PARTICLE_COUNT * 3);

           const particleSizes = new Float32Array(PARTICLE_COUNT);

           const particleColors = new Float32Array(PARTICLE_COUNT * 3);

           for (let i = 0; i < PARTICLE_COUNT; i++) {

               // Random position within bounds

               const x = Math.random() * 60 - 30;

               const y = Math.random() * 20 - 10;

               const z = Math.random() * 60 - 30;

               particlePositions[i * 3] = x;

               particlePositions[i * 3 + 1] = y;

               particlePositions[i * 3 + 2] = z;

               // Varying particle sizes

               particleSizes[i] = Math.random() * 0.5 + 0.1;

               // Red color with slight variations

               particleColors[i * 3] = 0.5 + Math.random() * 0.5; // Red

               particleColors[i * 3 + 1] = 0; // Green

               particleColors[i * 3 + 2] = 0; // Blue

           }

           particleGeometry.setAttribute('position', new THREE.BufferAttribute(particlePositions, 3));

           particleGeometry.setAttribute('size', new THREE.BufferAttribute(particleSizes, 1));

           particleGeometry.setAttribute('color', new THREE.BufferAttribute(particleColors, 3));

           // Create point material

           const particleMaterial = new THREE.PointsMaterial({

               color: 0xff0000,

               size: 0.5,

               transparent: true,

               opacity: 0.6,

               vertexColors: true

           });

           const particleSystem = new THREE.Points(particleGeometry, particleMaterial);

           scene.add(particleSystem);

           particles.push(particleSystem);

       }

       // CREATE 3D BUTTONS AS SYMBIOTIC INTERFACES

       function createSymbioticButtons() {

           // Create 3D button models to match the HTML links

           const buttonGeometry = new THREE.BoxGeometry(3, 3, 1);

           const buttonMaterial1 = new THREE.MeshBasicMaterial({

           scene.add(button1);

           // Second button

           const buttonMaterial2 = new THREE.MeshBasicMaterial({

               color: 0x330000,

               wireframe: true

           });

           const button2 = new THREE.Mesh(buttonGeometry, buttonMaterial2);

           button2.position.set(10, -3, 15);

           scene.add(button2);

       }

       // SYNAPSE FORMATION BETWEEN NEURONS

       function connectNeurons(neuronA, neuronB) {

           // Create axon geometry

           const points = [];

           points.push(new THREE.Vector3(neuronA.position.x, neuronA.position.y, neuronA.position.z));

           // Add intermediary points for realistic axon path

           const midX = (neuronA.position.x + neuronB.position.x) / 2;

           const midY = (neuronA.position.y + neuronB.position.y) / 2;

           const midZ = (neuronA.position.z + neuronB.position.z) / 2;

           // Random displacement to avoid straight lines

           const displaceY = (Math.random() * 4) - 2;

           points.push(new THREE.Vector3(midX, midY + displaceY, midZ));

           points.push(new THREE.Vector3(neuronB.position.x, neuronB.position.y, neuronB.position.z));

           // Create the axon fiber

           const axonGeometry = new THREE.BufferGeometry().setFromPoints(points);

           const axonMaterial = new THREE.LineBasicMaterial({

               color: 0x223344,

               transparent: true,

               opacity: 0.6

           });

           const axon = new THREE.Line(axonGeometry, axonMaterial);

           scene.add(axon);

       }

       // STIMULUS RESPONSE FUNCTIONS

       function stimulateSynapse(event) {

           // Convert user input to neural stimulation

           const rect = container.getBoundingClientRect();

           const mouseX = ((event.clientX - rect.left) / container.clientWidth) * 2 - 1;

           const mouseY = -((event.clientY - rect.top) / container.clientHeight) * 2 + 1;

           // Neural excitation based on receptor input

           camera.position.x = mouseX * 8;

           camera.lookAt(mouseX * 3, 0, 0);

       }

       function triggerActionPotential(event) {

           // Depolarize system

           synapticResponse.style.opacity = 1;

           synapticResponse.innerText = 'neural_cascade_' + Math.floor(Math.random() * 1000);

           // Create depolarization wave

           scene.background = new THREE.Color(AXON_COLOR);

           // Random neuron excitation

           nervousSystem.forEach(neuron => {

               neuron.material.color.set(AXON_COLOR);

               neuron.scale.multiplyScalar(1.5);

           });

           // Accelerate particles

           if (particles.length > 0) {

               const positions = particles[0].geometry.attributes.position.array;

               for (let i = 0; i < positions.length; i += 3) {

                   positions[i] *= 1.05;

                   positions[i + 1] *= 1.05;

                   positions[i + 2] *= 1.05;

               }

               particles[0].geometry.attributes.position.needsUpdate = true;

           }

           // Repolarization after delay

           setTimeout(() => {

               scene.background = new THREE.Color(MEMBRANE_COLOR);

               nervousSystem.forEach(neuron => {

                   neuron.material.color.set(SYNAPSE_COLOR);

                   neuron.scale.set(1, 1, 1);

               });

               // Reset particles

               if (particles.length > 0) {

                   const positions = particles[0].geometry.attributes.position.array;

                   for (let i = 0; i < positions.length; i += 3) {

                       positions[i] *= 0.95;

                       positions[i + 1] *= 0.95;

                       positions[i + 2] *= 0.95;

                   }

                   particles[0].geometry.attributes.position.needsUpdate = true;

               }

               // Hyperpolarization

               setTimeout(() => {

                   synapticResponse.style.opacity = 0;

               }, 300);

           }, 150);

       }

       function recalibrateVisualField() {

           camera.aspect = container.clientWidth / container.clientHeight;

           camera.updateProjectionMatrix();

           renderer.setSize(container.clientWidth, container.clientHeight);

       }

       // NEURAL OSCILLATION SYSTEM

       function oscillateNeuralSystem() {

           requestAnimationFrame(oscillateNeuralSystem);

           renderNeuralOutput();

       }

       function renderNeuralOutput() {

           syntheticTime += 0.01;

           // Simulate ion channel fluctuations

           scene.rotation.y = Math.sin(syntheticTime * 0.1) * 0.05;

           // Neuron oscillation patterns

           nervousSystem.forEach((neuron, i) => {

               neuron.rotation.x += 0.002;

               neuron.rotation.z += 0.003;

               // Simulate neural rhythm

               const oscillation = Math.sin(syntheticTime + i * 0.1) * 0.1;

               neuron.position.y += oscillation;

               // Return to homeostasis

               neuron.position.y *= 0.99;

           });

           // Update particle system motion

           if (particles.length > 0) {

               const positions = particles[0].geometry.attributes.position.array;

               for (let i = 0; i < positions.length; i += 3) {

                   // Brownian motion to simulate neurotransmitter diffusion

                   positions[i] += (Math.random() - 0.5) * 0.05;

                   positions[i + 1] += (Math.random() - 0.5) * 0.05;

                   positions[i + 2] += (Math.random() - 0.5) * 0.05;

                   // Contain particles within bounds

                   if (Math.abs(positions[i]) > 30) positions[i] *= 0.95;

                   if (Math.abs(positions[i + 1]) > 15) positions[i + 1] *= 0.95;

                   if (Math.abs(positions[i + 2]) > 30) positions[i + 2] *= 0.95;

               }

               particles[0].geometry.attributes.position.needsUpdate = true;

           }

           // Organic layer pulsation

           organicLayers.forEach((layer, i) => {

               // Simulate tissue respiration

               layer.rotation.x += Math.sin(syntheticTime * 0.2) * 0.0005;

               layer.rotation.z += Math.cos(syntheticTime * 0.15) * 0.0003;

               // Scale pulsation

               const pulsation = Math.sin(syntheticTime * 0.3 + i * 0.5) * 0.02 + 1;

               layer.scale.set(1, pulsation, 1);

           });

           renderer.render(scene, camera);

       }

       // NEURAL SYSTEM ACTIVATION

       initializeNeuralInterface();

       oscillateNeuralSystem();
