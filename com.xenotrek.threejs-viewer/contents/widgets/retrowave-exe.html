   function generateAsciiArt() {

     const techChars = 'xX0@#*+=:-.01>!|/\\[]{}$%&<>?;:~^`';

     let ascii = '';

     const rows = 30;

     const cols = 40;

     // Technical terms to randomly insert

     const techTerms = [

       'HACK', 'SYS', 'ROOT', 'NET', 'PWN',

       'EXE', 'CMD', 'GET', 'LOG', 'KEY',

       'SSH', 'GPU', 'CPU', 'RAM', 'EOF',

       'HEX', 'BIN', 'ASM', 'SQL', 'DNS',

       'DOS', 'WIN', 'USB', 'FTP', 'VPN'

     ];

     for (let i = 0; i < rows; i++) {

       // Random chance to insert a tech term instead of random chars

       if (Math.random() < 0.2) {

         const term = techTerms[Math.floor(Math.random() * techTerms.length)];

         const padding = ' '.repeat(Math.floor(Math.random() * (cols - term.length)));

         ascii += padding + term + ' '.repeat(cols - padding.length - term.length) + '\n';

       } else {

         for (let j = 0; j < cols; j++) {

           ascii += techChars[Math.floor(Math.random() * techChars.length)];

         }

         ascii += '\n';

       }

     }

     document.getElementById('ascii-art').textContent = ascii;

   }

       wireframeMaterial

     );

     capacitor.position.y = 0.14;

     // Add antenna

     const antenna = new THREE.Mesh(

       new THREE.CylinderGeometry(0.01, 0.01, 0.3, 4),

       wireframeMaterial

     );

     antenna.position.set(0, 0.3, 0);

     // Add wires

     for (let i = 0; i < 4; i++) {

       const wirePath = new THREE.CurvePath();

       const angle = (i / 4) * Math.PI * 2;

       const start = new THREE.Vector3(

         Math.cos(angle) * 0.1,

         0.12,

         Math.sin(angle) * 0.1

       );

       const end = new THREE.Vector3(

         Math.cos(angle) * 0.2,

         0.12,

         Math.sin(angle) * 0.2

       );

       const mid = new THREE.Vector3(

         Math.cos(angle) * 0.15,

         0.17,

         Math.sin(angle) * 0.15

       );

       const curve = new THREE.QuadraticBezierCurve3(start, mid, end);

       wirePath.add(curve);

       const wire = new THREE.Mesh(

         new THREE.TubeGeometry(curve, 8, 0.01, 4, false),

         wireframeMaterial

       );

       empGroup.add(wire);

     }

     empGroup.add(body, circuit, capacitor, antenna);

     return empGroup;

   }

   // Lockpick model

   function createLockpickModel() {

     const lockpickGroup = new THREE.Group();

     // Lock body

     const lockBody = new THREE.Mesh(

       new THREE.CylinderGeometry(0.2, 0.2, 0.3, 16),

       wireframeMaterial

     );

     lockBody.rotation.x = Math.PI / 2;

     // Lock shackle

     const curve = new THREE.CurvePath();

     const p1 = new THREE.Vector3(0, 0.15, -0.1);

     const p2 = new THREE.Vector3(0, 0.3, -0.1);

     const p3 = new THREE.Vector3(0, 0.3, 0.1);

     const p4 = new THREE.Vector3(0, 0.15, 0.1);

     curve.add(new THREE.LineCurve3(p1, p2));

     curve.add(new THREE.LineCurve3(p2, p3));

     curve.add(new THREE.LineCurve3(p3, p4));

     const shackle = new THREE.Mesh(

       new THREE.TubeGeometry(curve, 32, 0.03, 8, false),

       wireframeMaterial

     );

     // Lockpick tool

     const pick = new THREE.Mesh(

       new THREE.BoxGeometry(0.01, 0.3, 0.01),

       wireframeMaterial

     );

     pick.position.set(0, 0, 0.15);

     pick.rotation.x = Math.PI / 4;

     // Tension wrench

     const wrench = new THREE.Mesh(

       new THREE.BoxGeometry(0.01, 0.25, 0.03),

       wireframeMaterial

     );

     wrench.position.set(0.05, 0, 0.1);

     wrench.rotation.x = Math.PI / 4;

     wrench.rotation.z = Math.PI / 12;

     // Keyhole

     const keyhole = new THREE.Mesh(

       new THREE.TorusGeometry(0.05, 0.02, 8, 16, Math.PI),

       wireframeMaterial

     );

     keyhole.position.set(0, 0, 0.16);

     keyhole.rotation.x = Math.PI / 2;

     lockpickGroup.add(lockBody, shackle, pick, wrench, keyhole);

     return lockpickGroup;

   }

   // Update mesh based on current settings

   function updateMesh() {

     scene.remove(mesh);

     let modelGroup;

     switch(currentModel) {

       case 'gun':

         modelGroup = createGunModel();

         break;

       case 'server':

         modelGroup = createServerModel();

         break;

       case 'emp':

         modelGroup = createEMPModel();

         break;

       case 'lockpick':

         modelGroup = createLockpickModel();

         break;

       default:

         modelGroup = createGunModel();

     }

     // Apply materials based on current render mode

     modelGroup.traverse((object) => {

       if (object.isMesh) {

         switch(currentRenderMode) {

           case 'wireframe':

             object.material = wireframeMaterial;

             break;

           case 'solid':

             object.material = solidMaterial;

             break;

           case 'points':

             // For points mode, we need to replace the mesh with points

             const pointsGeom = object.geometry.clone();

             const points = new THREE.Points(pointsGeom, pointsMaterial);

             points.position.copy(object.position);

             points.rotation.copy(object.rotation);

             points.scale.copy(object.scale);

             object.parent.add(points);

             object.parent.remove(object);

             break;

         }

       }

     });

     mesh = modelGroup;

     scene.add(mesh);

   }

   // Create layered duplicates of the model for depth effect

   let layerMeshes = [];

   function updateLayerDepth() {

     // Remove existing layer meshes

     layerMeshes.forEach(layer => scene.remove(layer));

     layerMeshes = [];

     if (layerDepth <= 1) return;

     // Create new layer duplicates

     for (let i = 1; i < layerDepth; i++) {

       if (mesh) {

         const layerMesh = mesh.clone();

         // Scale and position the layer

         const scale = 1 + (i * 0.03);

         const offset = -i * 0.05;

         layerMesh.scale.set(scale, scale, scale);

         layerMesh.position.z = offset;

         // Make the layer semi-transparent

         layerMesh.traverse((object) => {

           if (object.isMesh) {

             const newMaterial = object.material.clone();

             newMaterial.transparent = true;

             newMaterial.opacity = 0.3 - (i * 0.02);

             object.material = newMaterial;

           }

         });

         scene.add(layerMesh);

         layerMeshes.push(layerMesh);

       }

     }

   }

   // Animation loop

   function animate() {

     requestAnimationFrame(animate);

     frame++;

     if (frame % 10 === 0) {

       const currentTime = performance.now();

       fps = Math.round(1000 / ((currentTime - lastTime) / 10));

       lastTime = currentTime;

       document.getElementById('fps-counter').textContent = `${fps} FPS`;

     }

     // Rotate the mesh

     if (mesh) {

       mesh.rotation.x += rotationSpeed;

       mesh.rotation.y += rotationSpeed * 1.3;

     }

     // Glitch effect (randomly distort geometry)

     if (glitchMode && frame % 30 === 0) {

       if (currentRenderMode !== 'points') {

         const positions = mesh.geometry.attributes.position.array;

         for (let i = 0; i < positions.length; i += 3) {

           if (Math.random() > 0.8) {

             positions[i] += (Math.random() - 0.5) * 0.1;

             positions[i+1] += (Math.random() - 0.5) * 0.1;

             positions[i+2] += (Math.random() - 0.5) * 0.1;

           }

         }

         mesh.geometry.attributes.position.needsUpdate = true;

       }

     }

     // Zoom effect

     if (zoomMode) {

       camera.position.z += 0.01 * zoomDirection;

       if (camera.position.z > 3.5) zoomDirection = -1;

       if (camera.position.z < 1.5) zoomDirection = 1;

     } else {

       // Gradually return to default position

       camera.position.z += (2.5 - camera.position.z) * 0.05;

     }

     // Animate lights

     pinkLight.position.x = 2 * Math.sin(Date.now() * 0.001);

     pinkLight.position.y = 2 * Math.cos(Date.now() * 0.001);

     blueLight.position.x = 2 * Math.cos(Date.now() * 0.0015);

     blueLight.position.z = 2 * Math.sin(Date.now() * 0.0015);

     // Animate sun

     sun.rotation.y += 0.005;

     // Scan effect

     if (scanMode && frame % 2 === 0) {

       scanLine += 3;

       if (scanLine > 230) scanLine = 0;

       const scanDiv = document.createElement('div');

       scanDiv.style.position = 'absolute';

       scanDiv.style.left = '0';

       scanDiv.style.top = `${scanLine}px`;

       scanDiv.style.width = '100%';

       scanDiv.style.height = '3px';

       scanDiv.style.backgroundColor = 'rgba(0, 255, 255, 0.5)';

       scanDiv.style.zIndex = '25';

       scanDiv.style.pointerEvents = 'none';

       scanDiv.style.boxShadow = '0 0 8px #00ffff';

       const oldScan = document.querySelector('.active-scan-line');

       if (oldScan) oldScan.remove();

       scanDiv.classList.add('active-scan-line');

       document.getElementById('scene-container').appendChild(scanDiv);

     }

     // Animate layer meshes

     if (layerMeshes.length > 0) {

       layerMeshes.forEach((layer, index) => {

         if (layer) {

           // Slightly offset rotation to create a trailing effect

           const offsetMultiplier = 0.8 - (index * 0.1);

           layer.rotation.x = mesh.rotation.x * offsetMultiplier;

           layer.rotation.y = mesh.rotation.y * offsetMultiplier;

           // Pulse effect on layers

           const pulseScale = 1 + (index * 0.03) + (Math.sin(Date.now() * 0.002 + index * 0.5) * 0.03);

           layer.scale.set(pulseScale, pulseScale, pulseScale);

         }

       });

     }

     renderer.render(scene, camera);

   }

   animate();
