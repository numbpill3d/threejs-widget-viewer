 <script>

   // TV static generator

   const staticCanvas = document.getElementById('static-canvas');

   const sctx = staticCanvas.getContext('2d');

   let staticIntensity = 0.25;

   let frameCount = 0;

   // generate noise using ImageData directly for speed

   const imgData = sctx.createImageData(520, 300);

   function drawStatic() {

     requestAnimationFrame(drawStatic);

     frameCount++;

     const d = imgData.data;

     for (let i = 0; i < d.length; i += 4) {

       // mostly dark with occasional bright flecks

       const v = Math.random() < staticIntensity ? Math.floor(Math.random() * 200) : Math.floor(Math.random() * 18);

       d[i] = v; d[i+1] = v; d[i+2] = v; d[i+3] = 255;

     }

     // occasional horizontal glitch bands

     if (Math.random() < 0.06) {

       const bandY = Math.floor(Math.random() * 295);

       const bandH = Math.floor(Math.random() * 4) + 1;

       for (let y = bandY; y < bandY + bandH && y < 300; y++) {

         const shift = Math.floor((Math.random() - 0.5) * 40);

         for (let x = 0; x < 520; x++) {

           const from = ((y * 520 + ((x + shift + 520) % 520)) * 4);

           const to   = (y * 520 + x) * 4;

           if (from >= 0 && from + 3 < d.length) {

             d[to] = d[from] + 60; d[to+1] = d[from+1]; d[to+2] = d[from+2]; d[to+3] = 255;

           }

         }

       }

     }

     sctx.putImageData(imgData, 0, 0);

     document.getElementById('frames-val').textContent = String(frameCount).padStart(4,'0').slice(-4);

   }

   drawStatic();

   // waveform

   const waveCanvas = document.getElementById('wave-canvas');

   const wctx = waveCanvas.getContext('2d');

   let waveOff = 0;

   function drawWave() {

     requestAnimationFrame(drawWave);

     wctx.clearRect(0,0,520,32);

     wctx.fillStyle = '#040404';

     wctx.fillRect(0,0,520,32);

     waveOff += 3;

     wctx.beginPath();

     wctx.strokeStyle = 'rgba(204,204,204,0.45)';

     wctx.lineWidth = 1;

     for (let x=0;x<520;x++) {

       const y = 16 + Math.sin((x+waveOff)*0.045)*5

                    + Math.sin((x+waveOff)*0.12)*2.5

                    + (Math.random()-0.5)*3.5;

       x===0 ? wctx.moveTo(x,y) : wctx.lineTo(x,y);

     }

     wctx.stroke();

     // red noise spikes

     for(let i=0;i<2;i++){

       const sx = Math.random()*520;

       wctx.beginPath();

       wctx.strokeStyle='rgba(255,34,0,0.5)';

       wctx.lineWidth=1;

       wctx.moveTo(sx,0); wctx.lineTo(sx,32);

       wctx.stroke();

     }

   }

   drawWave();

   // VHS fragment collage

   const seeds = [11,22,33,44,55,66,77,88,99,111,133,155,177,199,211,233,255,277];

   const activeFrags = [];

   let totalFrags = 0;

   function spawnFrag() {

     const layer = document.getElementById('collage-layer');

     const seed = seeds[Math.floor(Math.random() * seeds.length)];

     const w = 90 + Math.floor(Math.random() * 130);

     const h = 60 + Math.floor(Math.random() * 90);

     const x = Math.floor(Math.random() * (510 - w));

     const y = Math.floor(Math.random() * (290 - h));

     const glitchTop = Math.floor(Math.random() * 80) + 10;

     const glitchShift = Math.floor((Math.random()-0.5) * 12);

     const isRedShift = Math.random() < 0.3;

     const zIdx = Math.floor(Math.random() * 5) + 1;

     // sometimes clip only part of image (VHS tracking artifact)

     const clipTop = Math.random() < 0.35 ? Math.floor(Math.random() * 30) : 0;

     const clipBot = Math.random() < 0.25 ? Math.floor(Math.random() * 20) : 0;

     const el = document.createElement('div');

     el.className = 'vhs-frag' + (isRedShift ? ' red-shift' : '');

     el.style.cssText = `

       left:${x}px; top:${y}px;

       width:${w}px; height:${h}px;

       --glitch-top: ${glitchTop}%;

       --glitch-shift: ${glitchShift}px;

       z-index:${zIdx};

       clip-path: inset(${clipTop}px 0 ${clipBot}px 0);

     `;

     const img = document.createElement('img');

     img.src = `https://picsum.photos/seed/${seed}/${w}/${h}`;

     img.onerror = () => el.remove();

     el.appendChild(img);

     layer.appendChild(el);

     totalFrags++;

     document.getElementById('frags-count').textContent = 'FRAGS: ' + totalFrags;

     requestAnimationFrame(() => requestAnimationFrame(() => el.classList.add('visible')));

     const maxFrags = staticIntensity < 0.3 ? 4 : staticIntensity < 0.5 ? 6 : 10;

     const lifetime = 4000 + Math.random() * 3000;

     setTimeout(() => {

       el.style.transition = 'opacity 0.5s ease';

       el.style.opacity = '0';

       setTimeout(() => { el.remove(); const i=activeFrags.indexOf(el); if(i>-1) activeFrags.splice(i,1); }, 500);

     }, lifetime);

     activeFrags.push(el);

     while (activeFrags.length > maxFrags + 3) {

       const old = activeFrags.shift();

       if (old && old.parentNode) { old.style.transition='opacity 0.3s'; old.style.opacity='0'; setTimeout(()=>old.remove(),300); }

     }

   }

   let spawnTimer;

   function setStatic(level, btn) {

     document.querySelectorAll('.btn').forEach(b=>b.classList.remove('active'));

     btn.classList.add('active');

     staticIntensity = level==='low' ? 0.22 : level==='med' ? 0.42 : 0.72;

     clearInterval(spawnTimer);

     const rate = level==='low' ? 1800 : level==='med' ? 1100 : 500;

     spawnTimer = setInterval(spawnFrag, rate);

   }

   window.setStatic = setStatic;

   spawnTimer = setInterval(spawnFrag, 1800);

   setTimeout(spawnFrag, 400);

   // live readouts

   function pad(n) { return String(n).padStart(2,'0'); }

   const carriers = ['LOST','WEAK','FAINT','NOISE','LOST','LOST'];

   setInterval(() => {

     const n = new Date();

     document.getElementById('clock').textContent = `${pad(n.getHours())}:${pad(n.getMinutes())}:${pad(n.getSeconds())}`;

     document.getElementById('noise-val').textContent = '-' + Math.floor(35+Math.random()*25) + 'dB';

     document.getElementById('carrier-val').textContent = carriers[Math.floor(Math.random()*carriers.length)];

     document.getElementById('bloom-val').textContent = (Math.random()).toFixed(2);

     document.getElementById('sig-str').textContent = 'SIGNAL: ' + Math.floor(5+Math.random()*40) + '%';

     document.getElementById('ch-num').textContent = pad(Math.floor(Math.random()*12)+1);

   }, 1100);
