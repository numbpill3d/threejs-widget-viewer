   // three.js: crystalline octahedron with inner wire lattice

   const container = document.getElementById('void-canvas');

   const W = 500, H = 240;

   const renderer = new THREE.WebGLRenderer({ antialias: true, alpha: true });

   renderer.setSize(W, H);

   renderer.setClearColor(0x010808, 1);

   container.appendChild(renderer.domElement);

   const scene = new THREE.Scene();

   const camera = new THREE.PerspectiveCamera(55, W / H, 0.1, 100);

   camera.position.set(0, 0, 3.8);

   scene.add(new THREE.AmbientLight(0x003333, 3));

   const tealLight = new THREE.PointLight(0x00e5cc, 5, 10);

   tealLight.position.set(2, 1, 2);

   scene.add(tealLight);

   const purpleLight = new THREE.PointLight(0x9900ff, 4, 8);

   purpleLight.position.set(-2, -1, 1);

   scene.add(purpleLight);

   const shapes = [

     new THREE.OctahedronGeometry(1.0, 0),

     new THREE.IcosahedronGeometry(1.0, 0),

     new THREE.TetrahedronGeometry(1.0, 0),

   ];

   const shapeIdx = { OCT: 0, ICOS: 1, TETRA: 2 };

   let currentShape = 0;

   const mat = new THREE.MeshPhongMaterial({

     color: 0x00e5cc,

     emissive: 0x002222,

     transparent: true,

     opacity: 0.7,

     shininess: 200,

   });

   const mesh = new THREE.Mesh(shapes[0], mat);

   scene.add(mesh);

   // inner crystal

   const innerMat = new THREE.MeshPhongMaterial({

     color: 0x9900ff,

     emissive: 0x110022,

     transparent: true,

     opacity: 0.4,

     shininess: 300,

   });

   const innerMesh = new THREE.Mesh(new THREE.OctahedronGeometry(0.55, 0), innerMat);

   scene.add(innerMesh);

   // wire shell

   const wireMat = new THREE.MeshBasicMaterial({ color: 0x00e5cc, wireframe: true, transparent: true, opacity: 0.12 });

   const wireMesh = new THREE.Mesh(new THREE.OctahedronGeometry(1.08, 0), wireMat);

   scene.add(wireMesh);

   // orbit line rings

   const ringGeo = new THREE.RingGeometry(1.35, 1.37, 64);

   const ringMat = new THREE.MeshBasicMaterial({ color: 0x00e5cc, side: THREE.DoubleSide, transparent: true, opacity: 0.15 });

   const ring1 = new THREE.Mesh(ringGeo, ringMat);

   ring1.rotation.x = Math.PI / 2;

   scene.add(ring1);

   const ring2 = new THREE.Mesh(ringGeo, ringMat.clone());

   ring2.rotation.y = Math.PI / 4;

   scene.add(ring2);

   function cycleShape(btn) {

     document.querySelectorAll('.ctrl-btn').forEach(b => b.classList.remove('active'));

     btn.classList.add('active');

     const key = btn.textContent.trim();

     const idx = shapeIdx[key] ?? 0;

     mesh.geometry = shapes[idx];

   }

   window.cycleShape = cycleShape;

   function animate() {

     requestAnimationFrame(animate);

     const t = Date.now() * 0.001;

     mesh.rotation.x += 0.005;

     mesh.rotation.y += 0.009;

     innerMesh.rotation.x -= 0.007;

     innerMesh.rotation.z += 0.005;

     wireMesh.rotation.x = mesh.rotation.x * 0.5;

     wireMesh.rotation.y = mesh.rotation.y * 0.5;

     ring1.rotation.z += 0.003;

     ring2.rotation.x += 0.004;

     // pulse

     const p = 1 + 0.05 * Math.sin(t * 1.8);

     mesh.scale.set(p, p, p);

     tealLight.position.x = 2.5 * Math.sin(t * 0.6);

     tealLight.position.y = 1.5 * Math.cos(t * 0.4);

     purpleLight.position.x = -2 * Math.cos(t * 0.8);

     purpleLight.position.z = 2 * Math.sin(t * 0.5);

     // color shift

     mat.emissive.setHex(Math.random() > 0.99 ? 0x006655 : 0x002222);

     renderer.render(scene, camera);

   }

   let noiseOffset = 0;

   function drawNoise() {

     requestAnimationFrame(drawNoise);

     nctx.clearRect(0, 0, 500, 36);

     nctx.fillStyle = '#010808';

     nctx.fillRect(0, 0, 500, 36);

     // signal line
